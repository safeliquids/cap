163void cap_nv_append_value(NamedValues * nv, const TypedUnion value) {
164 static const size_t INIT_ALLOC = 1u;
165 if (!nv) {
166 return;
167 }
168 size_t alloc = nv -> mValueAlloc;
169 if (nv -> mValueCount >= alloc) {
170 alloc = alloc ? alloc * 2 : INIT_ALLOC;
171 nv -> mValues = (TypedUnion *) realloc(
172 nv -> mValues, alloc * sizeof(TypedUnion));
173 nv -> mValueAlloc = alloc;
174 }
175 nv -> mValues[nv -> mValueCount++] = value;
176}
