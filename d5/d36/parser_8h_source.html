589void cap_parser_destroy(ArgumentParser * parser) {
590 if (!parser) return;
591 if (parser -> mProgramName) {
592 free(parser -> mProgramName);
593 parser -> mProgramName = NULL;
594 }
595 delete_string_property(&(parser -> mDescription));
596 delete_string_property(&(parser -> mEpilogue));
597 delete_string_property(&(parser -> mCustomHelp));
598 for (size_t i = 0; i < parser -> mFlagCount; ++i) {
599 cap_flag_info_destroy(parser -> mFlags[i]);
600 }
601 for (size_t i = 0; i < parser -> mPositionalCount; ++i) {
602 cap_positional_info_destroy(parser -> mPositionals[i]);
603 }
604 free(parser -> mFlags);
605 free(parser -> mPositionals);
606 parser -> mFlags = NULL;
607 parser -> mPositionals = NULL;
608 parser -> mFlagCount = parser -> mFlagAlloc = 0u;
609 parser -> mPositionalCount = parser -> mPositionalAlloc = 0;
610
611 delete_string_property(&(parser -> mFlagPrefixChars));
612
613 if (parser -> mHelpFlagInfo) {
614 cap_flag_info_destroy(parser -> mHelpFlagInfo);
615 parser -> mHelpFlagInfo = NULL;
616 }
617 if (parser -> mFlagSeparatorInfo) {
618 cap_flag_info_destroy(parser -> mFlagSeparatorInfo);
619 parser -> mFlagSeparatorInfo = NULL;

920AddFlagError cap_parser_add_flag_noexit(
921 ArgumentParser * parser, const char * flag, DataType type,
922 int min_count, int max_count, const char * metavar,
923 const char * description)
924{
925 const char * const flag_prefix = parser -> mFlagPrefixChars;
926 if (!parser) {
927 return AFE_MISSING_PARSER;
928 }
929 if (!flag || !strlen(flag)) {
930 return AFE_MISSING_NAME;
931 }
932 if (!strchr(flag_prefix, *flag)) {
933 return AFE_INVALID_PREFIX;
934 }
935 // this also checks agains the help flag and flag separator if they exist
936 if (_cap_parser_find_flag(parser, flag)) {
937 return AFE_DUPLICATE;
938 }
939 if (min_count < 0) {
940 return AFE_MIN_COUNT_NEGATIVE;
941 }
942 if (max_count >= 0 && max_count < min_count) {
943 return AFE_MAX_COUNT_VIOLATION;
944 }
945 if (min_count == 0 && max_count == 0) {
946 return AFE_MAX_COUNT_ZERO;
947 }
948 if (parser -> mFlagCount >= parser -> mFlagAlloc) {
949 size_t alloc_size = parser -> mFlagAlloc;
950 alloc_size = alloc_size ? alloc_size * 2 : 1;
951 parser -> mFlagAlloc = alloc_size;
952 parser -> mFlags = (FlagInfo **) realloc(
953 parser -> mFlags, alloc_size * sizeof(FlagInfo *));
954 }
955 FlagInfo * new_flag = cap_flag_info_make(
956 flag, metavar, description, type, min_count, max_count);
957 parser -> mFlags[parser -> mFlagCount++] = new_flag;
958
959 return AFE_OK;
960}

1008void cap_parser_add_flag(
1009 ArgumentParser * parser, const char * flag, DataType type,
1010 int min_count, int max_count, const char * meta_var,
1011 const char * description)
1012{
1013 AddFlagError error = cap_parser_add_flag_noexit(
1014 parser, flag, type, min_count, max_count, meta_var, description);
1015 switch (error) {
1016 case AFE_OK:
1017 return;
1018 case AFE_MISSING_PARSER:
1019 fprintf(stderr, "cap: missing parser\n");
1020 break;
1021 case AFE_MISSING_NAME:
1022 fprintf(stderr, "cap: missing flag name");
1023 break;
1024 case AFE_INVALID_PREFIX:
1025 fprintf(
1026 stderr, "cap: invalid flag name - must begin with one"
1027 " of \"%s\"\n", parser -> mFlagPrefixChars);
1028 break;
1029 case AFE_DUPLICATE:
1030 fprintf(stderr, "cap: duplicate flag definition %s\n", flag);
1031 break;
1032 case AFE_MIN_COUNT_NEGATIVE:
1033 fprintf(
1034 stderr, "cap: min_count requirement must not be"
1035 " negative\n");
1036 break;
1037 case AFE_MAX_COUNT_VIOLATION:
1038 fprintf(
1039 stderr, "cap: max_count requirement must not be less than"
1040 " min_count\n");
1041 break;
1042 case AFE_MAX_COUNT_ZERO:
1043 fprintf(
1044 stderr, "cap: min_count and max_count cannot be both zero\n");
1045 break;
1046 default:
1047 assert(false && "unreachable in cap_parser_add_flag_noexit");
1048 }
1049 exit(-1);
1050}

1068 ArgumentParser * parser, const char * name, const char * alias)
1069{
1070 if (!parser) {

1072 }

1074 return AFAE_MISSING_NAME;

1083 if (!fi) {

1087 return AFAE_DUPLICATE_ALIAS;

1089
1090 // register the alias
1091 if (fi -> mAliasCount >= fi -> mAliasAlloc) {
1092 fi -> mAliasAlloc = fi -> mAliasAlloc ? 2 * fi -> mAliasAlloc : 1u;
1093 fi -> mAliases = (char **) realloc(fi -> mAliases, fi -> mAliasAlloc * sizeof(char *));
1094 }
1095 fi -> mAliases[fi -> mAliasCount++] = copy_string(alias);
1096 return AFAE_OK;

1167void cap_parser_set_help_flag(
1168 ArgumentParser * parser, const char * name, const char * description) {
1169 static const char * const DEFAULT_HELP_DESCRIPTION
1170 = "Display this help message and exit";
1171
1172 if (!parser) {
1173 return;
1174 }
1175 if (parser -> mHelpFlagInfo) {
1176 // name is identical -> there's nothing to do
1177 if (name && !strcmp(name, parser -> mHelpFlagInfo -> mName)) {
1178 return;
1179 }
1180 // now we un-configure the existing help flag
1181 cap_flag_info_destroy(parser -> mHelpFlagInfo);
1182 parser -> mHelpFlagInfo = NULL;
1183 }
1184
1185 // at this moment, the help flag is definitely not configured: either it
1186 // did not exist, or it just got removed.
1187
1188 // if the goal was to remove the help flag, we end here
1189 if (!name) {
1190 return;
1191 }
1192
1193 if (_cap_parser_find_flag(parser, name)) {
1194 fprintf(
1195 stderr, "cap: cannot add help flag '%s' because an identical flag"
1196 " already exists\n", name);
1197 exit(-1);
1198 }
1199 if (*name == '\0' || !strchr(parser -> mFlagPrefixChars, *name)) {
1200 fprintf(stderr, "cap: invalid flag name '%s'\n", name);
1201 exit(-1);
1202 }
1203 FlagInfo * fi = cap_flag_info_make(
1204 name, NULL, description ? description : DEFAULT_HELP_DESCRIPTION,
1205 DT_PRESENCE, 0, 1);
1206 parser -> mHelpFlagInfo = fi;
1207}

1253AddPositionalError cap_parser_add_positional_noexit(
1254 ArgumentParser * parser, const char * name, DataType type, bool required,
1255 bool variadic, const char * metavar, const char * description)
1256{
1257 if (!parser) {
1258 return APE_MISSING_PARSER;
1259 }
1260 if (!name || strlen(name) == 0) {
1261 return APE_MISSING_NAME;
1262 }
1263 if (type == DT_PRESENCE) {
1264 return APE_PRESENCE;
1265 }
1266 for (size_t i = 0; i < parser -> mPositionalCount; ++i) {
1267 const PositionalInfo * pi = parser -> mPositionals[i];
1268 if (!strcmp(pi -> mName, name)) {
1269 return APE_DUPLICATE;
1270 }
1271 }
1272 if (parser -> mPositionalCount >= 1u) {
1273 const PositionalInfo * last
1274 = parser -> mPositionals[parser -> mPositionalCount - 1u];
1275 if (last -> mVariadic) {
1276 return APE_ANYTHING_AFTER_VARIADIC;
1277 }
1278 if (required && !last -> mRequired) {
1279 return APE_REQUIRED_AFTER_OPTIONAL;
1280 }
1281 }
1282 if (parser -> mPositionalCount >= parser -> mPositionalAlloc) {
1283 size_t alloc_size = parser -> mPositionalAlloc;
1284 alloc_size = alloc_size ? alloc_size * 2 : 1;
1285 parser -> mPositionalAlloc = alloc_size;
1286 parser -> mPositionals = (PositionalInfo **) realloc(
1287 parser -> mPositionals, alloc_size * sizeof(PositionalInfo *));
1288 }
1289 PositionalInfo * new_positional = cap_positional_info_make(
1290 name, metavar, description, type, required, variadic);
1291 parser -> mPositionals[parser -> mPositionalCount++] = new_positional;
1292
1293 return APE_OK;
1294}

1389const char * cap_parser_get_program_name(
1390 const ArgumentParser * parser, const char * argv0) {
1391 if (parser -> mProgramName) {
1392 return parser -> mProgramName;
1393 }
1394 const char * program_name = argv0;
1395 const char * slash = strrchr(argv0, '/');
1396 if (slash) {
1397 program_name = slash + 1;
1398 }
1399#ifdef _WIN32
1400 const char * backslash = strrchr(argv0, '\\');
1401 if (backslash && backslash > slash) {
1402 program_name = backslash + 1;
1403 }
1404#endif
1405 return program_name;
1406}

1414void cap_parser_print_usage(
1415 const ArgumentParser * parser, FILE * file,
1416 const char * argv0) {
1417 if (!parser || !file) {
1418 return;
1419 }
1420
1421 if (!parser -> mEnableUsage) {
1422 return;
1423 }
1424 if (parser -> mCustomUsage) {
1425 fprintf(file, "%s\n", parser -> mCustomUsage);
1426 return;
1427 }
1428
1429 fprintf(file, "usage:\n");
1430 fprintf(file, "\t");
1431 fprintf(file, "%s", cap_parser_get_program_name(parser, argv0));
1432
1433 if (parser -> mHelpFlagInfo) {
1434 fprintf(file, " [%s]", _cap_get_shortest_flag_name(parser -> mHelpFlagInfo));
1435 }
1436
1437 for (size_t i = 0; i < parser -> mFlagCount; ++i) {
1438 const FlagInfo * fi = parser -> mFlags[i];
1439 fputc(' ', file);
1440 if (fi -> mMinCount == 0) {
1441 fputc('[', file);
1442 }
1443 fprintf(file, "%s", _cap_get_shortest_flag_name(fi));
1444 if (fi -> mType != DT_PRESENCE) {
1445 fprintf(file, " %s", cap_get_flag_metavar(fi));
1446 }
1447 if (fi -> mMinCount == 0) {
1448 fputc(']', file);
1449 }
1450 }
1451
1452 if (parser -> mPositionalCount > 0u && parser -> mFlagSeparatorInfo) {
1453 fprintf(file, " [%s]", _cap_get_shortest_flag_name(parser -> mFlagSeparatorInfo));
1454 }
1455 size_t optionals = 0u;
1456 for (size_t i = 0; i < parser -> mPositionalCount; ++i) {
1457 const PositionalInfo * pi = parser -> mPositionals[i];
1458 fputc(' ', file);
1459 if (!pi -> mRequired) {
1460 fputc('[', file);
1461 ++optionals;
1462 }
1463 if (pi -> mMetaVar) {
1464 fprintf(file, "%s", pi -> mMetaVar);
1465 continue;
1466 }
1467 fprintf(file, "<%s>", pi -> mName);
1468 }
1469 for (size_t i = 0; i < optionals; ++i) {
1470 fputc(']', file);
1471 }
1472 fputc('\n', file);

1596ParsedArguments * cap_parser_parse(
1597 ArgumentParser * parser, int argc, const char ** argv) {
1598 ParsingResult result = cap_parser_parse_noexit(parser, argc, argv);
1599 if (result.mError == PER_NO_ERROR) {
1600 return result.mArguments;
1601 }
1602 if (result.mError == PER_HELP) {
1603 cap_parser_print_usage(parser, stdout, *argv);
1604 putchar('\n');
1605 cap_parser_print_help(parser, stdout);
1606 exit(0);
1607 }
1608 fprintf(stderr, "%s: ", cap_parser_get_program_name(parser, *argv));
1609 switch (result.mError) {
1610 case PER_NOT_ENOUGH_POSITIONALS:
1611 fprintf(stderr, "not enough arguments");
1612 break;
1613 case PER_TOO_MANY_POSITIONALS:
1614 fprintf(stderr, "too many arguments");
1615 break;
1616 case PER_CANNOT_PARSE_POSITIONAL:
1617 fprintf(
1618 stderr, "cannot parse value '%s' for argument '%s'",
1619 result.mSecondErrorWord, result.mFirstErrorWord);
1620 break;
1621 case PER_UNKNOWN_FLAG:
1622 fprintf(stderr, "unknown flag '%s'", result.mFirstErrorWord);
1623 break;
1624 case PER_MISSING_FLAG_VALUE:
1625 fprintf(
1626 stderr, "missing value for flag '%s'", result.mFirstErrorWord);
1627 break;
1628 case PER_CANNOT_PARSE_FLAG:
1629 fprintf(
1630 stderr, "cannot parse value '%s' for flag '%s'",
1631 result.mSecondErrorWord, result.mFirstErrorWord);
1632 break;
1633 case PER_NOT_ENOUGH_FLAGS:
1634 fprintf(
1635 stderr, "not enough instances of flag '%s'",
1636 result.mFirstErrorWord);
1637 break;
1638 case PER_TOO_MANY_FLAGS:
1639 fprintf(
1640 stderr, "too many instances of flag '%s'",
1641 result.mFirstErrorWord);
1642 break;
1643 case PER_HELP:
1644 case PER_NO_ERROR:
1645 default:
1646 assert(false && "unreachable in parsing error checking");
1647
1648 }
1649 fprintf(stderr, "\n\n");
1650 cap_parser_print_usage(parser, stderr, *argv);
1651 exit(-1);
1652}

1758static OnePositionalParsingResult _cap_parser_parse_one_positional(
1759 const ArgumentParser * parser, const char * arg,
1760 size_t positional_index) {
1761 OnePositionalParsingResult res;
1762 res.mPositional = NULL;
1763 res.mWordsConsumed = 0;
1764 res.mError = OPPE_NO_ERROR;
1765
1766 if (positional_index >= parser -> mPositionalCount) {
1767 res.mError = OPPE_TOO_MANY;
1768 return res;
1769 }
1770 const PositionalInfo * posit_info
1771 = parser -> mPositionals[positional_index];
1772 res.mPositional = posit_info;
1773 if (!_cap_parse_word_as_type(arg, posit_info -> mType, &(res.mValue))) {
1774 res.mError = OPPE_CANNOT_PARSE;
1775 return res;
1776 }
1777
1778 res.mWordsConsumed = 1;
1779 return res;

1782static OneFlagParsingResult _cap_parser_parse_one_flag(
1783 const ArgumentParser * parser, int argc, const char * const * argv,
1784 int index) {
1785 OneFlagParsingResult result;
1786 result.mWordsConsumed = 0;
1787 result.mError = OFPE_NO_ERROR;
1788
1789 const char * arg = argv[index];
1790
1791 // 1. is this a flag that exists?
1792 const FlagInfo * flag_info = _cap_parser_find_flag(parser, arg);
1793 result.mFlag = flag_info;
1794 if (!flag_info) { // no such flag was found
1795 result.mError = OFPE_UNKNOWN_FLAG;
1796 return result;
1797 }
1798 // skip arg because it is being consumed right now
1799 ++index;
1800 ++result.mWordsConsumed;
1801
1802 // 3. check data type and try to parse it
1803 if (flag_info -> mType == DT_PRESENCE) {
1804 result.mValue = cap_tu_make_presence();
1805 return result;
1806 }
1807 // parse the next argument according to dtype
1808 if (index >= argc) {
1809 result.mError = OFPE_MISSING_FLAG_VALUE;
1810 return result;
1811 }
1812 const char * value_arg = argv[index];
1813 // This is a bit messy but it should work.
1814 // Generally, it is bad practice to use uninitialized objects.
1815 // What's even funnier is, I made factory functions for typed unions
1816 // but this code isn't directly using them lol.
1817 if (!_cap_parse_word_as_type(
1818 value_arg, flag_info -> mType, &(result.mValue))) {
1819 result.mError = OFPE_CANNOT_PARSE_FLAG;
1820 return result;
1821 }
1822 // very important! must skip extra the word that was consumed here
1823 ++index;
1824 ++result.mWordsConsumed;
1825 return result;
1826}
1827
1828static void _cap_parser_parse_flags_and_positionals(
1829 const ArgumentParser * parser, int argc, const char * const * argv,
1830 ParsingResult * result) {
1831 size_t positional_index = 0;
1832 int index = 1;
1833 bool positional_only = false;
1834
1835 while (index < argc) {
1836 const char * arg = argv[index];
1837
1838 if (positional_only || !strlen(arg)
1839 || !strchr(parser -> mFlagPrefixChars, arg[0])) {
1840 // positional
1841 OnePositionalParsingResult one_posit_res =
1842 _cap_parser_parse_one_positional(parser, arg, positional_index);
1843 const PositionalInfo * posit_info = one_posit_res.mPositional;
1844 switch (one_posit_res.mError) {
1845 case OPPE_NO_ERROR:
1846 break;
1847 case OPPE_TOO_MANY:
1848 result -> mError = PER_TOO_MANY_POSITIONALS;
1849 return;
1850 case OPPE_CANNOT_PARSE:
1851 result -> mError = PER_CANNOT_PARSE_POSITIONAL;
1852 result -> mFirstErrorWord = posit_info -> mName;
1853 result -> mSecondErrorWord = arg;
1854 return;
1855 default:
1856 assert(
1857 false && "unreachable in "
1858 "_cap_parser_parse_flags_and_positionals");
1859 }
1860 cap_pa_append_positional(
1861 result -> mArguments, posit_info -> mName,
1862 one_posit_res.mValue);
1863 if (!posit_info -> mVariadic) {
1864 // if the current argument is variadic, do not advance
1865 // positional_index. That way more words can be consumed by
1866 // this.
1867 ++positional_index;
1868 }
1869 index += one_posit_res.mWordsConsumed;
1870 continue;
1871 }
1872
1873 // try to parse a flag
1874 OneFlagParsingResult one_flag_res = _cap_parser_parse_one_flag(
1875 parser, argc, argv, index);
1876 const FlagInfo * parsed_flag = one_flag_res.mFlag;
1877 switch (one_flag_res.mError) {
1878 case OFPE_NO_ERROR:
1879 break;
1880 case OFPE_UNKNOWN_FLAG:
1881 result -> mError = PER_UNKNOWN_FLAG;
1882 result -> mFirstErrorWord = arg;
1883 return;
1884 case OFPE_MISSING_FLAG_VALUE:
1885 result -> mError = PER_MISSING_FLAG_VALUE;
1886 result -> mFirstErrorWord = arg;
1887 return;
1888 case OFPE_CANNOT_PARSE_FLAG:
1889 result -> mError = PER_CANNOT_PARSE_FLAG;
1890 result -> mFirstErrorWord = arg;
1891 result -> mSecondErrorWord
1892 = argv[index + one_flag_res.mWordsConsumed];
1893 return;
1894 default:
1895 assert(false && "unreachable in cap_parser_parse_noexit");
1896 }
1897 index += one_flag_res.mWordsConsumed;
1898 if (parsed_flag == parser -> mFlagSeparatorInfo) {
1899 // switch to positional-only mode
1900 positional_only = true;
1901 continue;
1902 }
1903 if (parsed_flag == parser -> mHelpFlagInfo) {
1904 result -> mError = PER_HELP;
1905 return;
1906 }
1907 // normal flag -> add its value to parsed_arguments
1908 cap_pa_add_flag(
1909 result -> mArguments, parsed_flag -> mName, one_flag_res.mValue);
1910 }
1911}

1913static FlagCountCheckResult _cap_parser_check_flag_counts(
1914 const ArgumentParser * parser,
1915 const ParsedArguments * parsed_arguments) {
1916
1917 // check min and max count requirements for flags
1918 for (size_t i = 0; i < parser -> mFlagCount; ++i) {
1919 const FlagInfo * flag_info = parser -> mFlags[i];
1920 size_t real_count = cap_pa_flag_count(
1921 parsed_arguments, flag_info -> mName);
1922 if (real_count < (unsigned int) flag_info -> mMinCount) {
1923 return (FlagCountCheckResult) {
1924 .mFlag = flag_info,
1925 .mCount = TOO_FEW
1926 };
1927 }
1928 if (flag_info -> mMaxCount >= 0
1929 && real_count > (unsigned int) flag_info -> mMaxCount) {
1930 return (FlagCountCheckResult) {
1931 .mFlag = flag_info,
1932 .mCount = TOO_MANY
1933 };
1934 }
1935 }
1936 return (FlagCountCheckResult) {
1937 .mFlag = NULL,
1938 .mCount = GOOD
1939 };
1940}

1942static void _cap_parser_check_flag_and_positional_counts(
1943 const ArgumentParser * parser, ParsingResult * result) {
1944 // positional argument presence is checked here
1945 //
1946 // if p_count is at least the number of positionals configured in the
1947 // parser, it is all good (NB that a case of too-many-arguments is caught
1948 // when actually parsing)
1949 //
1950 // if p_count is less, find the PositionalInfo of the first argument that
1951 // was not parsed. If it is required, fail.
1952 const size_t p_count = cap_nva_length(result -> mArguments -> mPositionals);
1953 if (p_count < parser -> mPositionalCount) {
1954 const PositionalInfo * first_not_parsed = parser -> mPositionals[p_count];
1955 if (first_not_parsed -> mRequired) {
1956 result -> mError = PER_NOT_ENOUGH_POSITIONALS;
1957 return;
1958 }
1959 }
1960 // required flag counts are checked using another function because we also
1961 // want to know the name of the flag
1962 FlagCountCheckResult count_check = _cap_parser_check_flag_counts(
1963 parser, result -> mArguments);
1964 switch (count_check.mCount) {
1965 case GOOD:
1966 break;
1967 case TOO_FEW:
1968 result -> mError = PER_NOT_ENOUGH_FLAGS;
1969 result -> mFirstErrorWord = count_check.mFlag -> mName;
1970 return;
1971 case TOO_MANY:
1972 result -> mError = PER_TOO_MANY_FLAGS;
1973 result -> mFirstErrorWord = count_check.mFlag -> mName;
1974 return;
1975 default:
1976 assert(false && "unreachable in cap_parser_parse_noexit");
1977 }
1978}
