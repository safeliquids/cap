195static void _cap_nva_append_item(
196 NamedValuesArray * nva, const char * name, const TypedUnion value)
197{
198 static const size_t INIT_ALLOC = 4u;
199 size_t alloc = nva -> mAlloc;
200 if (nva -> mCount >= alloc) {
201 alloc = alloc ? alloc * 2 : INIT_ALLOC;
202 nva -> mItems = (NamedValues **) realloc(
203 nva -> mItems, alloc * sizeof(NamedValues *));
204 nva -> mAlloc = alloc;
205 }
206 // since we know that name is not NULL, this will not create NULL
207 nva -> mItems[nva -> mCount++] = cap_nv_make(name, value);
208}
